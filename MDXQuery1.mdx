with 
 member [Measures].[ConverNullToZore_FataledTotal] as
 CoalesceEmpty([Measures].[Fataled Total],0)
 member [Measures].[ConverNullToZore_InfectedTotal] as
 CoalesceEmpty([Measures].[Infected Total],0)
 member [Measures].[ConverNullToZore_ResolvedTotal] as
 CoalesceEmpty([Measures].[Resolved Total],0)
select non empty crossjoin({[Measures].[ConverNullToZore_FataledTotal], 
[Measures].[ConverNullToZore_InfectedTotal], [Measures].[ConverNullToZore_ResolvedTotal]}, 
[Dim Phu].[Phu Name].[Phu Name]) on rows,
non empty [Dim Day].[Year].[Year] on columns
from [Case_Amount_Cube]
